<script>
  let gates = [];

  async function loadGates() {
    const res = await fetch("/api/gates");
    gates = await res.json();
    render();
  }

  async function toggleStatus(id) {
    const gate = gates.find(g => g.id === id);
    if (!gate) return;

    const newStatus = gate.status === "Operational" ? "Down" : "Operational";

    await fetch(`/api/gates/${id}/status`, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ status: newStatus })
    });

    await loadGates();
  }

  function render() {
    const container = document.querySelector(".grid");
    container.innerHTML = "";

    gates.forEach(g => {
      const isDown = g.status === "Down";

      container.innerHTML += `
        <div class="card">
          <div class="name">${g.name}</div>
          <div class="meta">Projeto: ${g.project}</div>
          <div class="meta">Grupo: ${g.group}</div>
          <div class="status ${isDown ? "down" : "operational"}">
            ${g.status}
          </div>
          <div class="actions">
            <button class="btn ${isDown ? "btn-green" : "btn-red"}"
              onclick="toggleStatus(${g.id})">
              ${isDown ? "Voltar Operational" : "Marcar Down"}
            </button>
          </div>
        </div>
      `;
    });
  }

  loadGates();
</script>
